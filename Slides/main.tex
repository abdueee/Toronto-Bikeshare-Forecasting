\begin{frame}
\frametitle{Project Overview}

\begin{itemize}
  \setlength\itemsep{0.6em}
  \item \textbf{Goal:} Forecast hour based bikeshare demand across Toronto.
  \item \textbf{Data:} Ridership (Jan 2025 -- Feb 2026) + station metadata + hourly weather; chronological train/val/test split.
  \item \textbf{Design choice:} K-means ($K=8$) on station coordinates cuts zero inflation from 77.7\% to 37.3\%.
  \item \textbf{Models:} Baselines with Linear Regression, improve with Random Forest, XGBoost and PyTorch MLP.
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Data Pipeline, EDA, and Feature Engineering}

{\footnotesize
\begin{itemize}
  \item \textbf{Ingestion:} 12 CSVs; trips outside 1\,min to 24\,hr dropped; aggregated to station-hour pickups with zeros filled.
  \item \textbf{Zero Inflation with Clustering:} 77.7\% station-hours are zero; K-means ($K=8$) on lat/lon groups 999 stations into 8 clusters, cutting zeros to \textbf{37.3\%}.
  \item \textbf{Features \& Split:} 12 features incl.\ \texttt{lag\_24h}; 2025 train $|$ Jan 2026 val $|$ Feb 2026 test.
\end{itemize}
}

\textbf{Metric} & \textbf{Train} & \textbf{Test} \\
\hline
MAE   & 30.71 & 11.18 \\
RMSE  & 93.52 & 33.01 \\
$R^2$ & 0.89  & 0.71  \\
\end{tabular}
}

\vspace{0.3cm}

{\footnotesize
\begin{itemize}
  \item \textbf{lag\_24h:} Standard forecasting practice; 24h chosen over 1h to avoid handing the model the answer.
  \item \textbf{Limits:} LR fits one slope per feature, cannot separate summer vs.\ winter rush hours; tree models and NNs close this gap.
\end{itemize}
}
