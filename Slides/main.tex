\begin{frame}
\frametitle{Slide 1}
\textbf{Project Overview}

\begin{itemize}
  \item Goal: forecast Toronto Bike Share demand at hourly resolution.
  \item Data scope: 2025 to early 2026 ridership, station metadata, and hourly weather.
  \item Core workflow: data cleaning $\rightarrow$ EDA $\rightarrow$ feature engineering $\rightarrow$ baseline model $\rightarrow$ tree-based models.
  \item Evaluation focus: MAE, RMSE, and $R^2$ on held-out test period.
\end{itemize}
\end{frame}


\begin{frame}
\frametitle{Slide 2}
\textbf{Data Cleaning and Ingestion}

\begin{itemize}
  \item Merged monthly ridership files for 2025--2026.
  \item Dropped rows with missing start station/time.
  \item Filtered trip duration to \texttt{60--86400} seconds.
  \item Aggregated to station-hour pickups and filled implicit zeros.
  \item Cleaned station metadata: \textbf{1031 stations}.
  \item Weather cleanup produced \textbf{0 nulls} in key columns.
\end{itemize}

\begin{block}{Output of This Stage}
Prepared consistent modeling files in \texttt{data/processed/} (hourly demand, clusters, weather, train/val/test).
\end{block}
\end{frame}
